\textbf{1.}
\begin{resposta}
\begin{lstlisting}
CREATE TABLE Professor (
 id SERIAL PRIMARY KEY,
 nome VARCHAR(100) NOT NULL,
 cpf CHAR(11) UNIQUE NOT NULL,
 titulacao VARCHAR(50),
 salario DECIMAL(10,2) CHECK (salario > 0),
 ativo BOOLEAN DEFAULT TRUE
);
\end{lstlisting}
\end{resposta}

\textbf{2.}
\begin{resposta}
\begin{lstlisting}
CREATE TABLE Veterinario (
 id SERIAL PRIMARY KEY,
 nome VARCHAR(100) NOT NULL,
 crmv VARCHAR(20) UNIQUE NOT NULL,
 especialidade VARCHAR(50),
 salario DECIMAL(10,2),
 data_contratacao DATE
);
\end{lstlisting}
\end{resposta}

\textbf{3.}
\begin{resposta}
\begin{lstlisting}
CREATE TABLE FornoIndustrial (
 id SERIAL PRIMARY KEY,
 modelo VARCHAR(50),
 temperatura_maxima INTEGER CHECK (temperatura_maxima > 0),
 capacidade_litros INTEGER,
 data_instalacao DATE,
 ativo BOOLEAN DEFAULT TRUE
);
\end{lstlisting}
\end{resposta}

\textbf{4.}
\begin{resposta}
\begin{lstlisting}
CREATE TABLE Caldeira (
 id SERIAL PRIMARY KEY,
 fabricante VARCHAR(100),
 pressao_maxima DECIMAL(8,2) CHECK (pressao_maxima > 0),
 volume DECIMAL(8,2),
 data_ultima_inspecao DATE
);
\end{lstlisting}
\end{resposta}

\textbf{8.}
\begin{resposta}
\begin{lstlisting}
SELECT nome, salario
FROM Professor
WHERE salario > 5000
ORDER BY salario DESC;
\end{lstlisting}
\end{resposta}
